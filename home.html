function renderRecentTrade() { var record = getLatestTradeRecord(); var card = document.getElementById('recentTradeCard'); var content = document.getElementById('tradeRecordContent'); var timeEl = document.getElementById('tradeTime'); if (!record) { // 无记录时显示占位文字，卡片保持可见 if (card) card.style.display = 'block'; if (timeEl) timeEl.textContent = '--'; if (content) content.innerHTML = '
暂无交易记录
'; return; } // 有记录时正常填充 if (card) card.style.display = 'block'; if (timeEl) timeEl.textContent = record.time; if (content) { var txHashDisplay = record.txHash ? '
交易哈希' + record.txHash.substring(0, 10) + '...' + record.txHash.substring(record.txHash.length - 8) + '
' : ''; content.innerHTML = '
交易对' + record.inputToken + ' → ' + record.outputToken + '
' + '
输入' + record.inputAmount + ' ' + record.inputToken + '
' + '
输出' + record.outputAmount + ' ' + record.outputToken + '
' + txHashDisplay; } }
